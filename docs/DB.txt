CREATE SEQUENCE PI_SEQ
INCREMENT BY 1
START WITH 100;

CREATE TABLE Plant(
id INT PRIMARY KEY,
name VARCHAR2(20),
join_date DATE
);

CREATE TABLE Plant_Info(
infoNum int,
id INT,
tem NUMBER,
lux NUMBER,
humi NUMBER,
time DATE,
CONSTRAINT FK_Plant_Info FOREIGN KEY(id)
REFERENCES Plant(id)
);
